function check_whether_complete(str)
    # String should start with "POLYGON ((" and end with "))"
    contains(str, "POLYGON ((") & contains(str, "))")

# Function to take the string containing the polygon coordinates and return a vector of tuples
function strcoords_to_tuplecoords(str)
    str = replace(str, "POLYGON ((", " ")                    # Remove first few characters
    str = replace(str, "))", "")                             # Remove last few characters
    str_vec = split(str, ',')                                # Split string into vector of substrings
    vec = Array(Tuple{Float64, Float64}, length(str_vec))    # Initialise array of tuples
    for i = eachindex(str_vec)
        s = split(strip(str_vec[i]), " ")                    # Clean and split the substring into two number strings
        vec[i] = (float(s[1]), float(s[2]))                  # Convert number strings into Float64 and collect in tuple
    end
    vec                                                      # Return vector of tuples
end;

# Identity i.e. using the lon/lat as x/y (also known as EPSG4326 or Equirectangular projection)
import Base.identity    # Need to import from Base to write another method called "identity"
identity(xLon, yLat) = Base.identity(xLon), Base.identity(yLat)

# Converts WGS84 coordinates to Web Mercator projection
function WGS84toWebMerc(xLon, yLat)

    # Check coordinates are in range
    @assert abs(xLon) <= 180
    @assert abs(yLat) < 90

    # Ellipsoid semi-major axis for WGS84 (metres)
    # This is the equatorial radius - the Polar radius is 6356752.0
    a = 6378137.0

    # Convert to radians
    λ = xLon * 0.01745329251994328    # λ = xLon * π / 180
    ϕ = yLat * 0.01745329251994328    # ϕ = yLat * π / 180

    # Convert to Web Mercator
    # Note that:
    # atanh(sin(ϕ)) = log(tan(π/4 + ϕ/2)) = 1/2 * log((1 + sin(ϕ)) / (1 - sin(ϕ)))
    x = a * λ
    y = a * atanh(sin(ϕ))

    return x, y
end;

# Function to take the array of points in the dataframe and convert to context units and then the Compose package polygon type
function convert_polygon_points(points, convertcoords)
    pts = Array(Tuple{Measures.Length{:cx, Float64}, Measures.Length{:cy, Float64}}, 0)
    for pt in points
        x, y = convertcoords(pt...)
        push!(pts, (x*cx, y*cy))
    end
    Compose.polygon(pts)
end

function draw_polygons(CAD_ID; convertcoords=identity,  
                       cad_line_width=0.40mm, cad_line_color=RGBA(1,1,0,0.9), cad_fill_color=nothing,  
                       struc_line_width=0.10mm, struc_line_color=RGBA(1,0,0,0.5), struc_fill_color=RGBA(1,0,0,0.5))
    
    # Find row number
    row = findfirst(df[:, :CAD_ID], CAD_ID)
    
    # Create canvas
    # Note that Compose draws from left top corner
    left, top = convertcoords(df[row, :MinX], df[row, :MinY])
    right, bottom = convertcoords(df[row, :MaxX], df[row, :MaxY])
    width, height = right-left, bottom-top
    dims = UnitBox(left, bottom, width, -height)
    canvas = context(units=dims)
    
    # Draw CAD
    canvas = compose(canvas, (context(), convert_polygon_points(df[row, :CAD_Polygon], convertcoords), 
                              linewidth(cad_line_width), stroke(cad_line_color), fill(cad_fill_color)))
    
    # Draw Structure
    for idx = row:row+df[row, :StructureCount]-1
        canvas = compose(canvas, (context(), convert_polygon_points(df[idx, :Structure_Polygon], convertcoords), 
                                  linewidth(struc_line_width), stroke(struc_line_color), fill(struc_fill_color)))
    end
     
    # Print some info
    println("Dataframe Row Number = $row")
    ratio = width / height
    println("Implied dimensions are:")
    println("Height = $height")
    println("Width = $width")
    println("Width / Height = $ratio")
    
    # Return the canvas
    return canvas
end;

function overlay(polygon, img)
    
    # Image dimensions
    width, height = size(img)
    
    # Create Cairo surface, draw polygon to surface, reinterpret as an array of RGB fixed point numbers
    surface = Cairo.CairoARGBSurface(zeros(UInt32, height, width))
    draw(PNG(surface), polygon)
    overlay = reinterpret(BGRA{FixedPointNumbers.UFixed{UInt8, 8}}, surface.data)
    
    # Create combined image
    outimg = similar(img)
    for i=1:width, j=1:height
        
        # Transparency value at pixel i,j
        α = overlay[i,j].alpha
        
        # Convert pixel value to same element type (RGB of FixedPointNumbers) as the image
        overlay_pixel = convert(eltype(img), overlay[i,j])
        
        # Merge the two pixels - weighting determined by the transparency
        β = one(α) - α
        outimg[i,j] = β*img[i,j] + α*overlay_pixel
    end
    
    # Return the combined image
    return outimg
end;
